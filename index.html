  const COOLDOWN_MS  = 3 * 60 * 60 * 1000;

  const TAP_EARN     = 25;

  const REDEEM_AT    = 1000;


  const { createClient } = supabase;

  const db = createClient(SUPABASE_URL, SUPABASE_KEY);


  // ── AUTH ──

  async function signInGoogle() {

    await db.auth.signInWithOAuth({

      provider: 'google',

      options: { redirectTo: window.location.href }

    });

  }


  async function getUser() {

    const { data: { user } } = await db.auth.getUser();

    return user;

  }


  // ── PROFILE / WALLET SETUP ──

  async function ensureWallet(userId) {

    const { data } = await db.from('wallets').select().eq('id', userId).single();

    if (!data) {

      await db.from('wallets').insert({ id: userId, balance_cents: 0, last_tap_at: null });

    }

  }


  async function saveProfile(userId, formData) {

    await db.from('profiles').upsert({

      id: userId,

      full_name: formData.name,

      phone: formData.phone,

      date_of_birth: formData.dob,

      waiver_signed: true,

      waiver_signed_at: new Date().toISOString()

    });

  }


  // ── TAP (server-side enforced cooldown) ──

  async function processTap(userId) {

    const { data: wallet } = await db

      .from('wallets').select().eq('id', userId).single();


    const now = Date.now();

    const lastTap = wallet.last_tap_at ? new Date(wallet.last_tap_at).getTime() : 0;


    if (now - lastTap < COOLDOWN_MS) {

      const remain = COOLDOWN_MS - (now - lastTap);

      return { ok: false, remain };

    }


    // Award 25 cents

    const newBalance = wallet.balance_cents + TAP_EARN;

    await db.from('wallets').update({

      balance_cents: newBalance,

      last_tap_at: new Date().toISOString()

    }).eq('id', userId);


    await db.from('transactions').insert({

      user_id: userId,

      type: 'earn',

      amount_cents: TAP_EARN,

      label: 'RFID tap'

    });


    return { ok: true, newBalance };

  }


  // ── REDEEM ──

  async function processRedeem(userId) {

    const { data: wallet } = await db

      .from('wallets').select().eq('id', userId).single();


    if (wallet.balance_cents < REDEEM_AT) return { ok: false };


    const newBalance = wallet.balance_cents - REDEEM_AT;

    await db.from('wallets').update({ balance_cents: newBalance }).eq('id', userId);


    const code = 'BM-' + Math.random().toString(36).toUpperCase().slice(2,6);

    await db.from('transactions').insert({

      user_id: userId,

      type: 'redeem',

      amount_cents: -REDEEM_AT,

      label: `Redeemed — code ${code}`

    });


    return { ok: true, code, newBalance };

  }


  // ── LOAD WALLET DATA ──

  async function loadWalletData(userId) {

    const [{ data: wallet }, { data: txns }] = await Promise.all([

      db.from('wallets').select().eq('id', userId).single(),

      db.from('transactions').select().eq('user_id', userId)

        .order('created_at', { ascending: false }).limit(20)

    ]);

    return { wallet, txns };

  }

        await refreshWalletUI(user.id, profile);

      }

    } else {

      showScreen('screen-tap');

    }

  });
